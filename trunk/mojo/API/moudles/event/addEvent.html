mojo('#target-2').addEvent('click', {
	args: 'test',
	fn: function(arg){
	    log.info('传入参数为: {}', arg)
	       .info('这是点击事件!')
	       .alert();
    }
})
.addEvent('click', {
	args: ['a', 'b'],
	fn: function(arg1, arg2){
		log.info('第一个: {}, 第二个: {}', arg1, arg2)
		   .info('这还是点击事件!')
		   .alert();
    }
});

mojo('#target-3').addEvent({
	'mouseover': function(){
		log.info('事件: {}', this.event.type)
		   .render(this.el);
	},
	
	'mouseout': {
		args: 'ok',
		fn: function(arg){
			log.info('事件: {}, 参数: {}', this.event.type, arg)
			.render(this.el);
		}
	} 
});
